  function sendFeedback() {

    if (!selectedRating) {

    const name = document.getElementById('reviewName').value.trim() || 'לקוח אנונימי';

    try {

      reviews = JSON.parse(localStorage.getItem('reviews')) || [];

      reviews = [];

    });

  function escapeHtml(str) {

    return String(str)

      .replaceAll('&', '&amp;')

      .replaceAll('<', '&lt;')

      .replaceAll('>', '&gt;')

      .replaceAll('"', '&quot;')

      .replaceAll("'", '&#039;');
